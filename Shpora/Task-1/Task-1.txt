#подключаем библиотеку,чтобы рассмотреть все перестановки заголовков
from itertools import permutations

#важно переписать строки сверху вниз
matrix = 'все строки из матрицы'.split()
graph = 'все ребра из графа'.split()

print(*range(1, 8)) #количество столбцов в матрице

for i in permutations('названия вершин'):
   if all(str(i.index(x) + 1) in matrix[i.index(y)] for x, y in graph):
       print(*i)
====================================================================================
Если в выводе результата программы отображаются несколько строк, то либо вы опечатались и стоит себя перепроверить, либо граф симметричный.
Если после перепроверки опечаток не найдено, значит граф точно симметричный
